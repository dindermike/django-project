server {
    listen 80;
    server_name django.mikedinder.com;

    return 301 https://$host$request_uri;

    access_log /var/log/nginx/django.access.log;
    error_log  /var/log/nginx/django.error.log;

    location = /static/mikedinder/images/favicon.ico { access_log off; log_not_found off; }

    location /static/ {
        root /var/www/django;
    }

    location /media/ {
        root /var/www/django;
        expires 30d;
        add_header Cache-Control "public, no-transform";
        access_log off;
    }

    location / {
        include proxy_params;
        proxy_pass http://unix:/run/gunicorn-django.sock;
    }
}

server {
    listen 443 ssl;
    server_name django.mikedinder.com;

    ssl_certificate {{ ssl_cert_path }};
    ssl_certificate_key {{ ssl_key_path }};
    include {{ ssl_options_path }};
    ssl_dhparam {{ ssl_dhparams_path }};

    # Set a maximum body size for file uploads
    client_max_body_size 100M;

    location = /static/mikedinder/images/favicon.ico { access_log off; log_not_found off; }
