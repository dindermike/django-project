server {
    listen 80;
    server_name django.mikedinder.com;

    return 301 https://$host$request_uri;

    access_log /var/log/nginx/django.access.log;
    error_log  /var/log/nginx/django.error.log;

    location = /favicon.ico { access_log off; log_not_found off; }

    location /static/ {
        root /var/www/django;
    }

    location /media/ {
        root /var/www/django;
    }

    location / {
        include proxy_params;
        proxy_pass http://unix:/run/gunicorn-django.sock;
    }
}

server {
    listen 443 ssl; # managed by Certbot
    server_name django.mikedinder.com;

    ssl_certificate /etc/letsencrypt/live/django.mikedinder.com/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/django.mikedinder.com/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

    location = /favicon.ico { access_log off; log_not_found off; }

    location /static/ {
        root /var/www/django;
    }

    location /media/ {
        root /var/www/django;
    }

    location / {
        include proxy_params;
        proxy_pass http://unix:/run/gunicorn-django.sock;
    }
}
