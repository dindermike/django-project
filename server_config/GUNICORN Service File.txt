[Unit]
Description=Gunicorn daemon for Django
Requires=gunicorn-django.socket
After=network.target

[Service]
User=www-data
Group=www-data
WorkingDirectory=/var/www/django
ExecStart=/var/www/django/venv/bin/gunicorn \
    --access-logfile /var/log/gunicorn/django-access.log \
    --error-logfile /var/log/gunicorn/django-error.log \
    --workers 3 \
    --bind unix:/run/gunicorn-django.sock \
    wsgi:application

[Install]
WantedBy=multi-user.target
